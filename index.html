⚙ Setup & Configuration
STEP 1 OF 4
Create your Google Sheet

Go to sheets.google.com and create a new blank spreadsheet. Name it SoleTrack Data. You don't need to add any columns — the script does that automatically.

STEP 2 OF 4
Add the Google Apps Script

In your Google Sheet, click Extensions → Apps Script. Delete any existing code. Paste the script below, then click Save (Ctrl+S).

// SoleTrack Google Apps Script // Paste this in Extensions → Apps Script → replace all → Save → Deploy const SHEET_NAME_SAMPLES = 'Samples'; const SHEET_NAME_SALES = 'Sales'; const SHEET_NAME_TOURS = 'Tours'; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); if (data.action === 'save') { writeSheet(ss, SHEET_NAME_SAMPLES, data.samples || []); writeSheet(ss, SHEET_NAME_SALES, data.sales || []); writeSheet(ss, SHEET_NAME_TOURS, data.tours || []); return ok('Saved'); } } catch(err) { return error(err.toString()); } } function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); return ok({ samples: readSheet(ss, SHEET_NAME_SAMPLES), sales: readSheet(ss, SHEET_NAME_SALES), tours: readSheet(ss, SHEET_NAME_TOURS) }); } catch(err) { return error(err.toString()); } } function writeSheet(ss, name, rows) { let sheet = ss.getSheetByName(name); if (!sheet) sheet = ss.insertSheet(name); sheet.clearContents(); if (!rows.length) return; const keys = Object.keys(rows[0]); const data = [keys, ...rows.map(r => keys.map(k => { const v = r[k]; if (Array.isArray(v)) return v.join(', '); return v == null ? '' : v; }))]; sheet.getRange(1, 1, data.length, data[0].length).setValues(data); } function readSheet(ss, name) { const sheet = ss.getSheetByName(name); if (!sheet) return []; const vals = sheet.getDataRange().getValues(); if (vals.length < 2) return []; const [hdrs, ...rows] = vals; return rows.map(r => { const obj = {}; hdrs.forEach((h, i) => { let v = r[i]; // Re-parse array fields if (['articles','colors','sampleIds'].includes(h) && typeof v === 'string' && v) v = v.split(', ').map(x => x.trim()).filter(Boolean); obj[h] = (v === '' || v === null || v === undefined) ? ( ['articles','colors','sampleIds'].includes(h) ? [] : '' ) : v; }); return obj; }); } function ok(data) { return ContentService.createTextOutput(JSON.stringify({ok:true, data})).setMimeType(ContentService.MimeType.JSON); } function error(msg) { return ContentService.createTextOutput(JSON.stringify({ok:false, msg})).setMimeType(ContentService.MimeType.JSON); }
